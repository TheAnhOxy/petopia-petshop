CREATE DATABASE IF NOT EXISTS petshop;
USE petshop;

SET FOREIGN_KEY_CHECKS = 0;

-- 1. Bảng users: Lưu thông tin user, admin, customer
CREATE TABLE users (
    user_id VARCHAR(10) PRIMARY KEY,  
    username VARCHAR(255) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(255),
    phone_number VARCHAR(255),
    role ENUM('CUSTOMER', 'ADMIN') DEFAULT 'CUSTOMER',
    loyalty_points INT DEFAULT 0,
    is_active TINYINT(1) DEFAULT 1,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    INDEX idx_email (email)
);

-- 2. Bảng addresses: Lưu địa chỉ user
CREATE TABLE addresses (
    address_id VARCHAR(10) PRIMARY KEY,  
    user_id VARCHAR(10),
    street VARCHAR(255) NOT NULL,
    ward VARCHAR(255),
    district VARCHAR(255),
    province VARCHAR(255) NOT NULL,
    country VARCHAR(255) DEFAULT 'Vietnam',
    is_default TINYINT(1) DEFAULT 0,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- 3. Bảng categories: Lưu trữ loại và giống thú cưng
CREATE TABLE categories (
    category_id VARCHAR(10) PRIMARY KEY, 
    name VARCHAR(255) NOT NULL,
    description VARCHAR(255),
    parent_id VARCHAR(10),
    image_url VARCHAR(255),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (parent_id) REFERENCES categories(category_id) ON DELETE SET NULL,
    INDEX idx_name (name)
);

-- 4. Bảng pets: Lưu thông tin thú cưng (thêm cột discountPrice)
CREATE TABLE pets (
    pet_id VARCHAR(10) PRIMARY KEY, 
    name VARCHAR(255) NOT NULL,
    description TEXT,
    category_id VARCHAR(10),
    age INT,
    gender ENUM('MALE', 'FEMALE', 'UNKNOWN') DEFAULT 'UNKNOWN', 
    price DOUBLE NOT NULL,
    discount_price DOUBLE DEFAULT NULL,
    health_status VARCHAR(255),
    vaccination_history TEXT,
    stock_quantity INT DEFAULT 1,
    status ENUM('AVAILABLE', 'SOLD', 'RESERVED', 'DRAFT') DEFAULT 'DRAFT',
    video_url VARCHAR(255),
    weight DOUBLE,
    height DOUBLE,
    color VARCHAR(255),
    fur_type ENUM('SHORT', 'LONG', 'CURLY', 'NONE', 'OTHER') DEFAULT 'OTHER',
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (category_id) REFERENCES categories(category_id) ON DELETE SET NULL
);

-- 5. Bảng pet_images: Lưu trữ hình ảnh của thú cưng
CREATE TABLE pet_images (
    image_id VARCHAR(10) PRIMARY KEY,  
    pet_id VARCHAR(10),
    image_url VARCHAR(255) NOT NULL,
    is_thumbnail TINYINT(1) DEFAULT 0,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id) ON DELETE CASCADE
);

-- 6. Bảng services: Lưu trữ các dịch vụ chăm sóc
CREATE TABLE services (
    service_id VARCHAR(10) PRIMARY KEY,  
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price DOUBLE NOT NULL,
    image_url VARCHAR(255),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6)
);

-- 7. Bảng wishlists: Danh sách yêu thích của người dùng
CREATE TABLE wishlists (
    wishlist_id VARCHAR(10) PRIMARY KEY,  
    user_id VARCHAR(10),
    pet_id VARCHAR(10),
    added_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id) ON DELETE CASCADE
);

-- 8. Bảng orders: Đơn hàng
CREATE TABLE orders (
    order_id VARCHAR(10) PRIMARY KEY, 
    user_id VARCHAR(10),
    address_id VARCHAR(10),
    total_amount DOUBLE NOT NULL, 
    status ENUM('PENDING', 'CONFIRMED', 'SHIPPED', 'DELIVERED', 'CANCELLED') DEFAULT 'PENDING',
    payment_status ENUM('UNPAID', 'PAID', 'FAILED', 'REFUNDED') DEFAULT 'UNPAID',
    desired_delivery_date DATE,
    note TEXT,
    phone_number VARCHAR(255) NOT NULL,
    shipping_fee DOUBLE DEFAULT 0,
    discount_amount DOUBLE DEFAULT 0,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL,
    FOREIGN KEY (address_id) REFERENCES addresses(address_id) ON DELETE SET NULL,
    INDEX idx_status (status)
);

-- 9. Bảng order_items: Các mục trong đơn hàng (thú cưng)
CREATE TABLE order_items (
    order_item_id VARCHAR(10) PRIMARY KEY,  
    order_id VARCHAR(10),
    pet_id VARCHAR(10),
    quantity INT NOT NULL DEFAULT 1,
    price_at_purchase DOUBLE NOT NULL,
    discount_applied DOUBLE DEFAULT 0,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id) ON DELETE CASCADE
);

-- 10. Bảng order_services: Dịch vụ kèm theo trong đơn hàng
CREATE TABLE order_services (
    order_service_id VARCHAR(10) PRIMARY KEY, 
    order_id VARCHAR(10),
    service_id VARCHAR(10),
    quantity INT NOT NULL DEFAULT 1,
    price_at_purchase DOUBLE NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (service_id) REFERENCES services(service_id) ON DELETE CASCADE
);

-- 11. Bảng pre_bookings: Đặt trước thú cưng
CREATE TABLE pre_bookings (
    booking_id VARCHAR(10) PRIMARY KEY,  
    user_id VARCHAR(10),
    pet_id VARCHAR(10),
    deposit_amount DOUBLE NOT NULL,
    expected_date DATE,
    status ENUM('PENDING', 'CONFIRMED', 'CANCELLED') DEFAULT 'PENDING',
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id) ON DELETE CASCADE
);

-- 12. Bảng payments: Thanh toán
CREATE TABLE payments (
    payment_id VARCHAR(10) PRIMARY KEY, 
    order_id VARCHAR(10),
    amount DOUBLE NOT NULL,
    payment_method ENUM('COD', 'BANK_TRANSFER', 'EWALLET', 'CREDIT_CARD') NOT NULL,
    payment_status ENUM('PENDING', 'SUCCESS', 'FAILED', 'REFUNDED') DEFAULT 'PENDING',
    transaction_id VARCHAR(255),
    payment_url VARCHAR(255),
    payer_info JSON,
    payment_date DATETIME(6),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    INDEX idx_transaction_id (transaction_id)
);

-- 13. Bảng delivery_providers: Nhà cung cấp vận chuyển
CREATE TABLE delivery_providers (
    provider_id VARCHAR(10) PRIMARY KEY,  
    name VARCHAR(100) NOT NULL,
    description TEXT,
    contact_phone VARCHAR(20),
    contact_email VARCHAR(100),
    api_endpoint VARCHAR(255),
    is_active TINYINT(1) DEFAULT 1,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6)
);

-- 14. Bảng deliveries: Thông tin vận chuyển
CREATE TABLE deliveries (
    delivery_id VARCHAR(10) PRIMARY KEY,  
    order_id VARCHAR(10) UNIQUE,
    provider_id VARCHAR(10),
    tracking_number VARCHAR(50) UNIQUE,
    shipping_method ENUM('STANDARD', 'EXPRESS', 'SAME_DAY', 'PICKUP') NOT NULL,
    shipping_fee DOUBLE NOT NULL,
    estimated_delivery_date DATETIME(6),
    actual_delivery_date DATETIME(6),
    delivery_status ENUM('PREPARING', 'SHIPPED', 'IN_TRANSIT', 'DELIVERED', 'RETURNED', 'FAILED') DEFAULT 'PREPARING',
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (provider_id) REFERENCES delivery_providers(provider_id) ON DELETE SET NULL,
    INDEX idx_tracking_number (tracking_number)
);

-- 15. Bảng delivery_history: Lịch sử trạng thái vận chuyển
CREATE TABLE delivery_history (
    history_id VARCHAR(10) PRIMARY KEY,
    delivery_id VARCHAR(10),
    status ENUM('PREPARING', 'SHIPPED', 'IN_TRANSIT', 'DELIVERED', 'RETURNED', 'FAILED') NOT NULL,
    description TEXT,
    location VARCHAR(255),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (delivery_id) REFERENCES deliveries(delivery_id) ON DELETE CASCADE
);

-- 16. Bảng vouchers: Voucher (mã giảm giá nhập thủ công)
CREATE TABLE vouchers (
    voucher_id VARCHAR(10) PRIMARY KEY, 
    code VARCHAR(255) UNIQUE NOT NULL,
    description VARCHAR(255),
    discount_type ENUM('PERCENTAGE', 'FIXED_AMOUNT') NOT NULL,
    discount_value DOUBLE NOT NULL,
    min_order_amount DOUBLE,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    max_usage INT,
    used_count INT DEFAULT 0,
    status ENUM('ACTIVE', 'INACTIVE') DEFAULT 'ACTIVE',
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    INDEX idx_code (code)
);

-- 17. Bảng promotions: Khuyến mãi
CREATE TABLE promotions (
    promotion_id VARCHAR(10) PRIMARY KEY,
    code VARCHAR(255) UNIQUE NOT NULL,
    description VARCHAR(255),
    promotion_type ENUM('FREESHIP', 'DISCOUNT', 'CASHBACK', 'BUNDLE') NOT NULL,
    discount_value DOUBLE NOT NULL,
    min_order_amount DOUBLE,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    max_usage INT,
    used_count INT DEFAULT 0,
    category_id VARCHAR(10),
    image_url VARCHAR(255),
    status ENUM('ACTIVE', 'INACTIVE') DEFAULT 'ACTIVE',
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (category_id) REFERENCES categories(category_id) ON DELETE SET NULL,
    INDEX idx_code (code)
);

-- 18. Bảng order_vouchers: Voucher áp dụng cho đơn hàng
CREATE TABLE order_vouchers (
    order_voucher_id VARCHAR(10) PRIMARY KEY, 
    order_id VARCHAR(10),
    voucher_id VARCHAR(10),
    discount_applied DOUBLE NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (voucher_id) REFERENCES vouchers(voucher_id) ON DELETE CASCADE
);

-- 19. Bảng order_promotions: Khuyến mãi áp dụng cho đơn hàng
CREATE TABLE order_promotions (
    order_promotion_id VARCHAR(10) PRIMARY KEY,  
    order_id VARCHAR(10),
    promotion_id VARCHAR(10),
    discount_applied DOUBLE NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE CASCADE,
    FOREIGN KEY (promotion_id) REFERENCES promotions(promotion_id) ON DELETE CASCADE
);

-- 20. Bảng reviews: Đánh giá và bình luận về thú cưng
CREATE TABLE reviews (
    review_id VARCHAR(10) PRIMARY KEY,  
    pet_id VARCHAR(10),
    user_id VARCHAR(10),
    rating INT CHECK (rating >= 1 AND rating <= 5),
    comment TEXT,
    image_url VARCHAR(255),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL
);

-- 21. Bảng articles: Tin tức và bài viết
CREATE TABLE articles (
    article_id VARCHAR(10) PRIMARY KEY,  
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    author_id VARCHAR(10),
    image_url VARCHAR(255),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (author_id) REFERENCES users(user_id) ON DELETE SET NULL
);

-- 22. Bảng article_comments: Bình luận cho bài viết
CREATE TABLE article_comments (
    comment_id VARCHAR(10) PRIMARY KEY,  
    article_id VARCHAR(10),
    user_id VARCHAR(10),
    content TEXT NOT NULL,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    updated_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6),
    FOREIGN KEY (article_id) REFERENCES articles(article_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE SET NULL
);

-- 23. Bảng notifications: Thông báo
CREATE TABLE notifications (
    notification_id VARCHAR(10) PRIMARY KEY,  
    user_id VARCHAR(10),
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    typeNote ENUM('ORDER_UPDATE', 'NEW_PET', 'VACCINATION_REMINDER', 'PROMOTION') NOT NULL,
    is_read TINYINT(1) DEFAULT 0,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);

-- 24. Bảng loyalty_transactions: Giao dịch điểm thưởng
CREATE TABLE loyalty_transactions (
    transaction_id VARCHAR(10) PRIMARY KEY,
    user_id VARCHAR(10),
    order_id VARCHAR(10),
    points_earned INT DEFAULT 0,
    points_used INT DEFAULT 0,
    description VARCHAR(255),
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (order_id) REFERENCES orders(order_id) ON DELETE SET NULL
);

-- 25. Bảng search_history: Lưu trữ lịch sử tìm kiếm
CREATE TABLE search_history (
    history_id VARCHAR(10) PRIMARY KEY,  
    user_id VARCHAR(10),
    querySearch VARCHAR(255) NOT NULL,
    created_at DATETIME(6) DEFAULT CURRENT_TIMESTAMP(6),
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE
);
